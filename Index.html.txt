  <script>
    // Définition du système source : Nord Sahara 1959 / Voirol Unifié Nord Algérie (EPSG:30791)
    proj4.defs(
      "EPSG:30791",
      "+proj=lcc +lat_1=36 +lat_2=33 +lat_0=35 +lon_0=2.7 +x_0=500000 +y_0=300000 +ellps=clrk80ign +pm=paris +towgs84=-73,246,227,0,0,0,0 +units=m +no_defs"
    );

    const src = "EPSG:30791";
    const dst = "WGS84";

    const els = {
      x: document.getElementById("x"),
      y: document.getElementById("y"),
      z: document.getElementById("z"),
      lat: document.getElementById("lat"),
      lon: document.getElementById("lon"),
      googleEarth: document.getElementById("googleEarth"),
      altitude: document.getElementById("altitude"),
      mapLink: document.getElementById("mapLink"),
      status: document.getElementById("status"),
      convertBtn: document.getElementById("convertBtn"),
      copyBtn: document.getElementById("copyBtn"),
      resetBtn: document.getElementById("resetBtn")
    };

    function setStatus(message, type = "") {
      els.status.className = `status ${type}`.trim();
      els.status.textContent = message;
    }

    function formatNum(value, decimals = 6) {
      return Number(value).toFixed(decimals);
    }

    function toDMS(decimal, isLat) {
      const abs = Math.abs(decimal);
      const deg = Math.floor(abs);
      const minFloat = (abs - deg) * 60;
      const min = Math.floor(minFloat);
      const sec = ((minFloat - min) * 60).toFixed(2);
      const hemi = isLat ? (decimal >= 0 ? "N" : "S") : (decimal >= 0 ? "E" : "O");
      return `${deg}° ${min}′ ${sec}″ ${hemi}`;
    }

    function convert() {
      const x = parseFloat(els.x.value);
      const y = parseFloat(els.y.value);
      const z = els.z.value.trim() === "" ? null : parseFloat(els.z.value);

      if (!Number.isFinite(x) || !Number.isFinite(y)) {
        setStatus("Veuillez saisir des valeurs numériques valides pour X et Y.", "error");
        return;
      }

      try {
        const [lon, lat] = proj4(src, dst, [x, y]);
        const latTxt = formatNum(lat, 6);
        const lonTxt = formatNum(lon, 6);
        const googleEarthTxt = `${latTxt}, ${lonTxt}`;

        els.lat.textContent = `${latTxt}  (${toDMS(lat, true)})`;
        els.lon.textContent = `${lonTxt}  (${toDMS(lon, false)})`;
        els.googleEarth.textContent = googleEarthTxt;
        els.altitude.textContent = Number.isFinite(z) ? `${z} m` : "Non renseignée";

        const mapsUrl = `https://www.google.com/maps?q=${encodeURIComponent(googleEarthTxt)}`;
        els.mapLink.innerHTML = `<a class="link" href="${mapsUrl}" target="_blank" rel="noopener noreferrer">Ouvrir dans Google Maps</a>`;

        setStatus("Conversion effectuée avec succès.", "ok");
      } catch (error) {
        console.error(error);
        setStatus("Erreur de conversion. Vérifiez les valeurs saisies.", "error");
      }
    }

    async function copyGoogleEarth() {
      const text = els.googleEarth.textContent;
      if (!text || text === "—") {
        setStatus("Aucun résultat à copier. Lancez d’abord la conversion.", "error");
        return;
      }

      try {
        await navigator.clipboard.writeText(text);
        setStatus("Coordonnées copiées dans le presse-papiers.", "ok");
      } catch {
        setStatus("Impossible de copier automatiquement. Copiez le texte manuellement.", "error");
      }
    }

    function resetForm() {
      els.x.value = "";
      els.y.value = "";
      els.z.value = "";
      els.lat.textContent = "—";
      els.lon.textContent = "—";
      els.googleEarth.textContent = "—";
      els.altitude.textContent = "—";
      els.mapLink.textContent = "—";
      setStatus("");
      els.x.focus();
    }

    els.convertBtn.addEventListener("click", convert);
    els.copyBtn.addEventListener("click", copyGoogleEarth);
    els.resetBtn.addEventListener("click", resetForm);

    [els.x, els.y, els.z].forEach(input => {
      input.addEventListener("keydown", (e) => {
        if (e.key === "Enter") convert();
      });
    });

    convert();
  </script>
